import { BookElasticModule, BookRedisModule } from '../../commons/modules';

import { BookService } from './book.abstract';

@Module({
    imports: [BookElasticModule, BookRedisModule],
    providers: [
        {
            provide: BookService,
            useClass: BookServiceImpl,
        },
    ],
    exports: [BookService],
})
export class BookModule {}

export { BookService };
